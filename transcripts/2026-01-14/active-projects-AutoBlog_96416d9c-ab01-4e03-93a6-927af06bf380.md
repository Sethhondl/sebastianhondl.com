import numpy as np
import matplotlib.pyplot as plt

# Problem: dy/dt = -2y + 1, y(0) = 0

def analytical_solution(t):
    """Exact solution: y(t) = 0.5(1 - exp(-2t))"""
    return 0.5 * (1 - np.exp(-2 * t))

def euler_method(f, t0, y0, h, t_final):
    """Euler's method for solving ODE"""
    t = np.arange(t0, t_final + h, h)
    y = np.zeros_like(t)
    y[0] = y0
    
    for i in range(len(t) - 1):
        y[i + 1] = y[i] + h * f(t[i], y[i])
    
    return t, y

def rk4_method(f, t0, y0, h, t_final):
    """4th-order Runge-Kutta method"""
    t = np.arange(t0, t_final + h, h)
    y = np.zeros_like(t)
    y[0] = y0
    
    for i in range(len(t) - 1):
        k1 = f(t[i], y[i])
        k2 = f(t[i] + h/2, y[i] + h*k1/2)
        k3 = f(t[i] + h/2, y[i] + h*k2/2)
        k4 = f(t[i] + h, y[i] + h*k3)
        y[i + 1] = y[i] + (h/6) * (k1 + 2*k2 + 2*k3 + k4)
    
    return t, y

def f(t, y):
    """ODE function: dy/dt = -2y + 1"""
    return -2 * y + 1
